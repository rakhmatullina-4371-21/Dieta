CREATE TABLE PATIENT_CARDS
(
         id_card SERIAL PRIMARY KEY,
         id_patient INT REFERENCES PATIENT(id_patient),
         id_employee INT REFERENCES EMPLOYEE(id_employee),
         daily_calories NUMERIC,
         daily_protein NUMERIC,
         daily_fats NUMERIC,
         daily_carbohydrates NUMERIC,
         start_diet DATE,
         finish_diet DATE,
         id_activity_levels INT REFERENCES ACTIVITY_LEVELS (id_activity_levels)
         activ boolean default false ;
);


CREATE TABLE MENU
(
         id_position SERIAL PRIMARY KEY,
         id_card INT REFERENCES PATIENT_CARDS (id_card),
         dish TEXT NOT NULL,
         calories NUMERIC,
         protein NUMERIC,
         fats NUMERIC,
         carbohydrates NUMERIC
);

CREATE TABLE MEALS
(

         id_meals SERIAL PRIMARY KEY,
         date DATE NOT NULL, 
         id_position INT REFERENCES MENU (id_position)
);
CREATE TABLE PATIENT_DIAGNOSES
(
         id_card INT REFERENCES PATIENT_CARDS (id_card),
         id_diagnosis INT REFERENCES DIAGNOSES (id_diagnosis)
);


CREATE TABLE  PATIENT_INDICATORS
(
         id_card_indicator SERIAL PRIMARY KEY,
         id_card INT REFERENCES PATIENT_CARDS (id_card),         
         id_indicator INT REFERENCES INDICATORS (id_indicator),            
         value_indicator  TEXT NOT NULL,
         result BOOLEAN DEFAULT FALSE    
         date_indicator DATE NOT NULL  
);
